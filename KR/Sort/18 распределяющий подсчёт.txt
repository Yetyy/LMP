 void sort( int *array, int *out, int *count, int size, int u, int v )
    {
        int i, j;
 
        for( i = 0; i < size; i++ )
            count[i] = 0;
 
        for( i = 0; i < size; i++ )
            count[array[i] - u]++;
 
        for( i = 1; i <= v - u; i++ )
            count[i] += count[i-1];
 
        for( i = size - 1; i >= 0; i-- )
        {
            j = count[array[i]];
            out[j] = array[i];
            count[array[i]]--;
        }
    }
